使用树状数组的前提：运算具有可合并性。

模板题目：[P3372 【模板】线段树 1](https://www.luogu.com.cn/problem/P3372)

题解：深进P86

板子：

```cpp
struct SegmentTree {
    int n;
    #define lson(u) ((u) << 1)
    #define rson(u) ((u) << 1 | 1)
    #define getLR() int L = t[u].l, R = t[u].r
    struct Node {
        int l, r;
        LL val_sum;
        LL tag_add;
        Node(
            int _l = 0, int _r = 0,
            LL _val_sum = 0,
            LL _tag_add = 0
        ) : //TODO: defalut value
        l(_l), r(_r), val_sum(_val_sum), tag_add(_tag_add) {}
    } t[MAXN << 2];

    Node friend operator+ (Node a, Node b) { // merge two nodes
        Node c;
        c.val_sum = a.val_sum + b.val_sum;
        return c;
    }

    void pushup(int u) {
        Node c = t[lson(u)] + t[rson(u)];
        t[u].val_sum = c.val_sum; //TODO: copy
    }
    void build(LL a[], int u, int L, int R) {
        t[u].l = L; t[u].r = R;
        if (L == R) {
            t[u].val_sum = a[L]; //TODO: init
            return;
        }
        int M = (L + R) >> 1;
        build(a, lson(u), L, M); build(a, rson(u), M + 1, R);
        pushup(u);
    }

    bool InRange(int L, int R, int l, int r) { // [L, R] in [l, r]?
        return (l <= L) && (R <= r);
    }
    bool OutofRange(int L, int R, int l, int r) { // [L, R] out of [l, r]?
        return (L > r) || (R < l);
    }

    void maketag(int u, LL tag_add) { //TODO: maketag
        getLR();
        int len = R - L + 1;
        t[u].tag_add += tag_add;
        t[u].val_sum += len * tag_add;
    }

    void pushdown(int u) {
        maketag(lson(u), t[u].tag_add);
        maketag(rson(u), t[u].tag_add);
        t[u].tag_add = 0; //TODO: remove the tag
    }

    Node query(int u, int l, int r) {
        getLR();
        if (InRange(L, R, l, r))
            return t[u];
        else if (!OutofRange(L, R, l, r)) {
            int M = (L + R) >> 1;
            pushdown(u);
            return query(lson(u), l, r) + query(rson(u), l, r);
        } else return Node();
    }

    void update(int u, int l, int r, LL k, int op) {
        getLR();
        if (InRange(L, R, l, r)) {
            if (op == 1)
                maketag(u, k);
        } else if (!OutofRange(L, R, l, r)) {
            pushdown(u);
            update(lson(u), l, r, k, op); update(rson(u), l, r, k, op);
            pushup(u);
        }
    }

    void build(LL a[], int _n) {
        n = _n;
        build(a, 1, 1, n);
    }
    void update(int l, int r, LL k, int op) {
        update(1, l, r, k, op);
    }
    Node query(int l, int r) {
        return query(1, l, r);
    }
};
```

使用：

```cpp
#include <bits/stdc++.h>
#define endl '\n'
using namespace std;
typedef long long LL;
typedef unsigned long long uLL;

const int MAXN = 1e5 + 5;


SegmentTree s;
LL a[MAXN];
int main() {
    int n, m;
    n = read(); m = read();
    for (int i = 1; i <= n; i++)
        a[i] = read();
    s.build(a, n);
    int op, x, y, k;
    while (m--) {
        op = read(); x = read(); y = read();
        if (op == 1) {
            k = read();
            s.update(x, y, k, op);
        } else if (op == 2)
            write(s.query(x, y).val_sum, '\n');
    }
    return 0;
}
```
