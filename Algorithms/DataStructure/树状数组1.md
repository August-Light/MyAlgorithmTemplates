模板题目：[P3374 【模板】树状数组 1](https://www.luogu.com.cn/problem/P3374)

板子：

```cpp
struct fenwick {
    int c[MAXN];
    int find(int x) {
        int ans = 0;
        for (int i = x; i; i -= lowbit(i))
            ans += c[i];
        return ans;
    }
    void modify(int x, int k) {
        for (int i = x; i < MAXN; i += lowbit(i))
            c[i] += k;
    }
};
```

使用：

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
typedef unsigned long long uLL;
const int MAXN = 5e5 + 5;
int lowbit(int x) {
    return x & -x;
}
int n, m, a[MAXN];


fenwick f;
int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        f.modify(i, a[i]);
    }
    while (m--) {
        int op, x, y;
        cin >> op >> x >> y;
        if (op == 1)
            f.modify(x, y);
        else
            cout << f.find(y) - f.find(x - 1) << endl;
    }
    return 0;
}
```
