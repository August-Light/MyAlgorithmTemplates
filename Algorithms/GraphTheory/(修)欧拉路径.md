模板题目：[P7771 【模板】欧拉路径](https://www.luogu.com.cn/problem/P7771)

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
typedef unsigned long long uLL;
const int MAXN = 1e5 + 5;
const int MAXM = 2e5 + 5;
int n, m;
vector<int> G[MAXN];
int in[MAXN], out[MAXN]; //入度和出度
int del[MAXN];
stack<int> st;
void dfs(int k) {
    for (int i = del[k]; i < G[k].size(); i = del[k]) {
        del[k] = i + 1;
        dfs(G[k][i]);
    }
    st.push(k);
}
bool solve(int n) {
    int S = 0, T = 0;
    for (int i = 1; i <= n; i++) {
        if (in[i] != out[i]) {
            if (out[i] == in[i] + 1) {
                if (S)
                    return false;
                S = i;
            } else if (in[i] == out[i] + 1) {
                if (T)
                    return false;
                T = i;
            } else return false;
        }
    }
    if (!S)
        S = T = 1;
    dfs(S);
    return true;
}
int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        out[u]++;
        in[v]++;
        G[u].push_back(v);
    }
    for (int i = 1; i <= n; i++)
        sort(G[i].begin(), G[i].end());
    if (!solve(n))
        puts("No");
    else
        while (!st.empty())
            printf("%d ", st.top()), st.pop();
    return 0;
}
```
