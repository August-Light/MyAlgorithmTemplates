struct Edge {
    int to, val, nxt;
};
struct Graph {
    int head[MAXN], cnt;
    Edge e[MAXM];
    void init() {
        memset(head, 0, sizeof(head));
        memset(e, 0, sizeof(e));
        cnt = 0;
    }
    void addedge(int u, int v, int w) {
        e[++cnt].to = v;
        e[cnt].val = w;
        e[cnt].nxt = head[u];
        head[u] = cnt;
    }
};
