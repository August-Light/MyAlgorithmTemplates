模板题目：[P1395 会议](https://www.luogu.com.cn/problem/P1395)

题解：P140

板子：

```cpp
void getCenter(int u, int fa) {
    siz[u] = 1, f[u] = 0;
    for (int i = 0; i < vec[u].size(); i++) {
        int v = vec[u][i];
        if (v == fa)
            continue;
        getCenter(v, u);
        siz[u] += siz[v];
        f[u] = max(f[u], siz[v]);
    }
    f[u] = max(f[u], n - siz[u]);
    if (f[u] < f[center] || (f[u] == f[center] && u < center))
        center = u;
}
```

使用：

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
typedef unsigned long long uLL;
const int MAXN = 1e6 + 5;
const int INF = INT_MAX;
int n;
int f[MAXN], siz[MAXN], dep[MAXN];
vector<int> vec[MAXN];
int center, ans;
void getCenter(int u, int fa);
void bfs() {
    queue<int> q;
    q.push(center);
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int i = 0; i < vec[u].size(); i++) {
            int v = vec[u][i];
            if (dep[v] || v == center)
                continue;
            dep[v] = dep[u] + 1;
            ans += dep[v];
            q.push(v);
        }
    }
}
int main() {
    cin >> n;
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        vec[u].push_back(v);
        vec[v].push_back(u);
    }
    center = 0, f[0] = INF;
    getCenter(1, 0);
    bfs();
    cout << center << " " << ans << endl;
    return 0;
}
```
