模板题目：[P3389 【模板】高斯消元法](https://www.luogu.com.cn/problem/P3389)

题解：深进P362

板子：

```cpp
void GaussElimination() {
    int cur = 0;
    for (int j = 0; j < n; j++) {
        int t;
        for (t = cur; t < n; t++)
            if (fabs(a[t][j]) > eps)
                break;
        if (t == n)
            continue;
        for (int i = j; i <= n; i++)
            swap(a[t][i], a[cur][i]);
        for (int i = n; i >= j; i--)
            a[cur][i] /= a[cur][j];
        for (int i = 0; i < n; i++)
            if (i != cur)
                for (int k = n; k >= j; k--)
                    a[i][k] -= a[cur][k] * a[i][j];
        cur++;
    }
    if (cur < n) {
        for (int i = cur; i < n; i++)
            if (fabs(a[i][n]) > eps) {
                puts("No Solution"); //无解
                return;
            }
        puts("No Solution"); //无穷多解
    } else
        for (int i = 0; i < n; i++)
            printf("%.2lf\n", a[i][n]); //输出解
}
```
