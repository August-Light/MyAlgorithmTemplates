[P1349 广义斐波那契数列](https://www.luogu.com.cn/problem/P1349)

$$
\begin{bmatrix}
p & q \\
1 & 0
\end{bmatrix}
\begin{bmatrix}
a_2 \\
a_1
\end{bmatrix}
$$

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
typedef unsigned long long uLL;
const int MAXN = 5;
const int L = 2;
int MOD;

int main() {
    int p, q, a1, a2, n;
    cin >> p >> q >> a1 >> a2 >> n >> MOD;
    if (n == 1)
        cout << a1 << endl;
    else if (n == 2)
        cout << a2 << endl;
    else {
        Matrix A, B;

        A.M[1][1] = p;  A.M[1][2] = q;
        A.M[2][1] = 1;  A.M[2][2] = 0;

        B.M[1][1] = a2;
        B.M[2][1] = a1;

        Matrix Ans = expow(A, n - 2) * B;
        cout << Ans.M[1][1] << endl;
    }
    return 0;
}
```
