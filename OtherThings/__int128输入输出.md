板子：

```cpp
typedef __int128 i128;
i128 read();
void write(i128 x, char c = '\0');
```

```cpp
i128 read() {
    char c = getchar();
    i128 x = 0;
    bool f = 0;
    for (; !isdigit(c); c = getchar()) f ^= !(c ^ 45);
    for (; isdigit(c); c = getchar()) x = (x << 1) + (x << 3) + (c ^ 48);
    if (f) x = -x;
    return x;
}
void write(i128 x, char c) {
    if (x < 0) putchar('-'), x = -x;
    if (x > 9) write(x / 10);
    putchar(x % 10 + '0');
    if (c != '\0') putchar(c);
}
```
